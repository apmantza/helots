def fetch_facts_oth(conn: sqlite3.Connection, leis_str: str, items_str: str, min_period: str) -> pd.DataFrame:
    """Fetch facts_oth rows for given LEIs and item IDs, joined with institution names."""
    query = f"""
        SELECT f.lei, COALESCE(i.short_name, i.commercial_name) as name, f.period, f.item_id, f.amount
        FROM facts_oth f
        JOIN institutions i ON f.lei = i.lei
        WHERE f.lei IN ({leis_str})
          AND f.item_id IN ({items_str})
          AND f.period >= '{min_period}'
    """
    return pd.read_sql(query, conn)
