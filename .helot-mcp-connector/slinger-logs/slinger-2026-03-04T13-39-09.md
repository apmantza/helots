private writeEvent(event: Record<string, any>): void {

    const eventsFile = join(this.governor.config.stateDir, 'events.jsonl');
    appendFileSync(eventsFile, JSON.stringify({ ts: new Date().toISOString(), ...event }) + '\n');

}
